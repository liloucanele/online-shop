    const product = localProducts[idx];

    boxFlavors.push({ name: product.name, price: product.price, quantity: qty });
    totalFlavorsCountInBox += qty;
    boxFlavorsCost += (product.price * qty);

    cart.push({ boxName: boxName, boxExtra: boxExtra, flavors: boxFlavors, flavorsCost: boxFlavorsCost, totalQty: totalFlavorsCountInBox, boxTotalCost: boxFlavorsCost + boxExtra });

    function calculateTotal() { let subtotal = 0; cart.forEach(box => { subtotal += box.boxTotalCost; }); document.getElementById('subtotalDisplay').innerText = `$${subtotal}`; const shipMethod = document.getElementById('shipMethod').value; let finalTotal = subtotal; if(shipMethod === 'delivery') finalTotal += shippingFee; if (appliedCoupon) { const minSpendRequirement = Number(appliedCoupon.min_spend) || 0; if (subtotal < minSpendRequirement) { appliedCoupon = null; const msgElement = document.getElementById('couponMessage'); msgElement.innerText = `✕ 組合變動，金額未達門檻 ${minSpendRequirement} 元，已自動取消折抵。`; msgElement.className = "text-xs md:text-sm font-bold mt-1.5 text-orange-500"; } } let discountAmount = 0; const couponDisplay = document.getElementById('couponDiscountDisplay'); if (appliedCoupon && subtotal > 0) { if (appliedCoupon.type === 'minus') { discountAmount = Number(appliedCoupon.value) || 0; } else if (appliedCoupon.type === 'discount') { discountAmount = Math.round(subtotal * (1 - Number(appliedCoupon.value))); } if (discountAmount > finalTotal) discountAmount = finalTotal; finalTotal -= discountAmount; couponDisplay.querySelector('span').innerText = `-$${discountAmount}`; couponDisplay.classList.remove('hidden'); } else { couponDisplay.classList.add('hidden'); } document.getElementById('totalDisplay').innerText = `$${finalTotal}`; return finalTotal; }
    window.applyCouponCode = function(isUserClick = false) {
    const inputVal = document.getElementById('couponInput').value.trim().toUpperCase();
    const msgElement = document.getElementById('couponMessage');
    if (inputVal === "") {
    appliedCoupon = null;
    msgElement.classList.add('hidden');
    calculateTotal();
    return;
    }
    let subtotal = 0;
    cart.forEach(box => { subtotal += box.boxTotalCost; });
    const match = localCoupons.find(c => c.code === inputVal);
    msgElement.classList.remove('hidden');
    if (match) {
    const minSpendRequirement = Number(match.min_spend) || 0;
    if (subtotal < minSpendRequirement) {
    appliedCoupon = null;
    msgElement.innerText = `✕ 未達消費門檻！此折扣碼最低消費需滿 ${minSpendRequirement} 元。`;
    msgElement.className = "text-xs md:text-sm font-bold mt-1.5 text-orange-500";
    } else {
    appliedCoupon = match;
    msgElement.innerText = `✓ 成功套用折扣碼！(${match.type === 'minus' ? '現折 ' + match.value + ' 元' : '享 ' + (match.value * 10) + ' 折'})`;
    msgElement.className = "text-xs md:text-sm font-bold mt-1.5 text-emerald-600";
    }
    } else {
    appliedCoupon = null;
    msgElement.innerText = "✕ 折扣碼無效或已過期。";
    msgElement.className = "text-xs md:text-sm font-bold mt-1.5 text-red-500";
    }
    calculateTotal();
    }
